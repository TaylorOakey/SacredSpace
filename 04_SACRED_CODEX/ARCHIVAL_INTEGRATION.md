## STEP 4 — BUILD THE INBOX PROCESSOR SCRIPT

Create `/mnt/d/SacredSpace_OS/02_COUNCIL_GROVE/inbox_processor.py`:

```python
#!/usr/bin/env python3
"""
INBOX PROCESSOR — S∆CR3D ∆RCHIV∆L
Watches the local INBOX/ for new .md files and:
1. Adds metadata header if missing
2. Routes to correct pillar folder
3. Ingests to ChromaDB
4. Updates SQLite manifest
5. Logs to Redis stream

Run: python3 inbox_processor.py
Or: import and call process_inbox() from Hermes
"""
import os, json, sqlite3, hashlib
from pathlib import Path
from datetime import datetime

VAULT   = Path("/mnt/d/01_VAULT/SacredSpace_Vault")
INBOX   = VAULT / "INBOX"
DB_PATH = Path("/mnt/d/SacredSpace_OS/05_MEMORY_ENGINE/sacredspace.db")

METADATA_TEMPLATE = """---
TITLE: {title}
DATE_CREATED: {date}
LAST_UPDATED: {date}
ORIGIN_NODE: {origin}
AUTHOR_SEAT: Taylor (∆∆∆O∆K3YTREE∆∆∆)
STATUS: DISTILLED
PILLAR: {pillar}
TAGS:
RELATED_ARTIFACTS:
CANON_STATUS: DISTILLED — pending Canon Gate
SECURITY_LEVEL: INTERNAL
SUMMARY:
---

"""

PILLAR_KEYWORDS = {
    "09_SACRED_MARKET":    ["revenue", "money", "cashflow", "etsy", "pod", "grant", "profit", "sales"],
    "04_SACRED_CODEX":     ["lore", "grama", "skry", "codex", "gematria", "arcana", "sigil", "canon"],
    "02_COUNCIL_GROVE":    ["protocol", "council", "hermes", "handoff", "governance", "ritual"],
    "03_NEURAL_FOREST":    ["model", "llm", "ollama", "chromadb", "vector", "embedding", "pipeline"],
    "06_AGENT_LAYER":      ["agent", "icaris", "elias", "aurora", "asher", "iris", "kethras"],
    "08_LEARNING_PATH":    ["maestro", "study", "learning", "course", "rite", "aas"],
    "07_SOCIAL_MOTHERSHIP":["brand", "social", "content", "instagram", "post", "creative"],
    "05_MEMORY_ENGINE":    ["memory", "sqlite", "redis", "database", "schema"],
}

def detect_pillar(content: str, filename: str) -> str:
    text = (content + " " + filename).lower()
    scores = {p: sum(1 for kw in kws if kw in text)
              for p, kws in PILLAR_KEYWORDS.items()}
    best = max(scores, key=scores.get)
    return best if scores[best] > 0 else "00_SACRED_SPINE"

def add_metadata(content: str, path: Path, pillar: str) -> str:
    if content.strip().startswith("---"):
        return content  # Already has metadata
    title = path.stem.replace("_", " ").replace("-", " ").title()
    header = METADATA_TEMPLATE.format(
        title=title,
        date=datetime.now().strftime("%Y-%m-%d"),
        origin="Council Grove",
        pillar=pillar
    )
    return header + content

def security_scan(content: str, filename: str) -> list:
    """Scan for sensitive data patterns."""
    import re
    flags = []
    patterns = {
        "api_key":   r"(?i)(api[_-]?key|bearer|token)\s*[:=]\s*\S{20,}",
        "password":  r"(?i)(password|passwd|pwd)\s*[:=]\s*\S+",
        "aws_key":   r"AKIA[0-9A-Z]{16}",
        "ssh_key":   r"-----BEGIN .* PRIVATE KEY-----",
    }
    for name, pattern in patterns.items():
        if re.search(pattern, content):
            flags.append(name)
    return flags

def process_file(md_path: Path):
    content = md_path.read_text(encoding="utf-8", errors="replace")
    filename = md_path.name

    # Security scan
    flags = security_scan(content, filename)
    if flags:
        dest = INBOX / "REVIEW_REQUIRED" / filename
        dest.parent.mkdir(parents=True, exist_ok=True)
        md_path.rename(dest)
        print(f"⚠ SECURITY: {filename} → REVIEW_REQUIRED/ [{', '.join(flags)}]")
        return

    # Detect pillar
    pillar = detect_pillar(content, filename)

    # Add metadata
    enriched = add_metadata(content, md_path, pillar)

    # Route to pillar
    dest = VAULT / pillar / filename
    dest.parent.mkdir(parents=True, exist_ok=True)
    dest.write_text(enriched, encoding="utf-8")

    # Ingest to ChromaDB
    try:
        import chromadb
        client = chromadb.HttpClient(host="localhost", port=8001)
        col    = client.get_or_create_collection("sacredspace")
        col.upsert(
            ids=[str(dest.relative_to(VAULT))],
            documents=[enriched],
            metadatas=[{
                "pillar": pillar, "source": "inbox_processor",
                "ts": datetime.now().isoformat()
            }]
        )
        print(f"✓ ChromaDB: {filename}")
    except Exception as e:
        print(f"  ChromaDB skip: {e}")

    # Log to SQLite
    try:
        conn = sqlite3.connect(str(DB_PATH))
        conn.execute("""
            CREATE TABLE IF NOT EXISTS inbox_log (
                id INTEGER PRIMARY KEY, filename TEXT, pillar TEXT,
                dest TEXT, ts TEXT, hash TEXT
            )
        """)
        file_hash = hashlib.sha256(enriched.encode()).hexdigest()[:16]
        conn.execute(
            "INSERT INTO inbox_log (filename, pillar, dest, ts, hash) VALUES (?,?,?,?,?)",
            (filename, pillar, str(dest), datetime.now().isoformat(), file_hash)
        )
        conn.commit()
        conn.close()
    except Exception as e:
        print(f"  SQLite skip: {e}")

    # Remove from inbox
    md_path.unlink()
    print(f"✓ Routed: {filename} → {pillar}/")

def process_inbox():
    """Process all .md files in INBOX/UNSORTED/ and INBOX/RAW_EXTRACTS/"""
    count = 0
    for folder in [INBOX / "UNSORTED", INBOX / "RAW_EXTRACTS"]:
        folder.mkdir(parents=True, exist_ok=True)
        for md_file in folder.glob("*.md"):
            print(f"\nProcessing: {md_file.name}")
            process_file(md_file)
            count += 1
    return count

if __name__ == "__main__":
    print("∆∆∆ S∆CR3D ∆RCHIV∆L — INBOX PROCESSOR ∆∆∆")
    n = process_inbox()
    print(f"\n✓ Processed {n} files")
    print("In lakesh alakin. ∆")
```

## STEP 5 — REGISTER inbox_process AS MCP TOOL

Append to hermes_mcp.py:

```python
@mcp.tool()
async def inbox_process() -> str:
    """
    Run the S∆CR3D ∆RCHIV∆L inbox processor.
    Scans INBOX/UNSORTED/ and INBOX/RAW_EXTRACTS/ for new .md files.
    Routes each file to its correct pillar, adds metadata, ingests to ChromaDB.
    Returns count of files processed.
    """
    import importlib.util
    spec = importlib.util.spec_from_file_location(
        "inbox_processor",
        "/mnt/d/SacredSpace_OS/02_COUNCIL_GROVE/inbox_processor.py"
    )
    mod = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(mod)
    count = mod.process_inbox()
    return json.dumps({"status": "complete", "processed": count,
                       "inbox": "/mnt/d/01_VAULT/SacredSpace_Vault/INBOX/"})
```
