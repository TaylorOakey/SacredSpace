4. API Bridge Pattern: Context Injection

`~/api/bodhilyra.ts`

```ts
import db from '../db';

export async function getBodhilyraContext(projectId?: string) {
  const context: string[] = [];
  
  if (projectId) {
    const project = await db('projects').where({ id: projectId }).first();
    const actions = await db('actions')
      .where({ project_id: projectId, is_done: false })
      .select('description');
    
    context.push(`Project: ${project.name}`);
    context.push(`Open Actions: ${actions.map(a => a.description).join(', ')}`);
  }
  
  // Add recent high-priority items
  const recentEntries = await db('entries')
    .orderBy('created_at', 'desc')
    .limit(3)
    .join('realms', 'entries.realm_id', 'realms.id')
    .select('entries.title', 'realms.name as realm');
    
  context.push(`Recent: ${recentEntries.map(e => `${e.title} [${e.realm}]`).join('; ')}`);
  
  return context.join(' | ');
}
```
